00058 # 
00059 #     Prepare data for neural network (assemble and normalize). Note: the polyfit
00060 #     coefficients and normalized row-wise, NOT column-wise as usual. Within a 
00061 #     row each block of 3 coefficients is normalized individually
00062 #     
00063 #     Used: 4 blocks of node cff1, cff2, cff3 (second order fit)
00064 #           Vmag
00065 #           Max - Min for normalized magnitutes
00066 #           chi2 from polyfit
00067 #     
00068 def prepdata(options, arr, cff, shuffle = True, 
00069              normsubtract = None, normdevide = None):
00070     nrrows = np.shape(arr)[0]
00071     
00072     # randomize order
00073     if (shuffle == True):
00074         order = range(nrrows)
00075         np.random.shuffle(order)
00076         arr = arr[order, :]
00077         cff = cff[order, :]
00078 
00079     nrinputs = 20
00080     newnorm = False
00081     alld = np.zeros((nrrows, nrinputs))
00082     if (normsubtract == None) or (normdevide == None):
00083         normsubtract = np.zeros(nrinputs)
00084         normdevide   = np.ones(nrinputs)
00085         newnorm = True
00086     allt = np.zeros((nrrows, len(options.classes)))
00087     allnames = []
00088     for i in xrange(nrrows):
00089         allnames.append(arr[i]['id'])
00090         # normalize coefficients per row
00091         coeffs = np.asarray(tuple(cff[i]))
00092         coeffs[3:6]   = (coeffs[3:6] - coeffs[3:6].mean()) / \
00093                            coeffs[3:6].std()
00094         coeffs[7:10]  = (coeffs[7:10] - coeffs[7:10].mean()) / \
00095                            coeffs[7:10].std()
00096         std = coeffs[11:14].std()
00097         if (std > 1e-6):
00098             coeffs[11:14] = (coeffs[11:14] - coeffs[11:14].mean()) / std
00099         std = coeffs[15:18].std()
00100         if (std > 1e-6):
00101             coeffs[15:18] = (coeffs[15:18] - coeffs[15:18].mean()) / std
00102                            
00103 
00104         # copy knots and normalized coeffs to data table
00105         alld[i, :16] = coeffs[2:]
00106         
00107         # add period, magnitude, difference of normalized magnitudes, chi2
00108         alld[i, 16]  = np.log10(arr[i]['period'])
00109         alld[i, 17]  = arr[i]['vmag']
00110         alld[i, 18]  = arr[i]['fmax'] - arr[i]['fmin']
00111         alld[i, 19]  = arr[i]['chi2']
00112         
00113         # make the target value line
00114         maketarget(options.classes, arr[i]['varcls'], allt[i])
00115 
00116     # normalize vmag
00117     if (newnorm == True):
00118         normsubtract[17] = alld[:, 17].mean()
00119         normdevide[17]   = alld[:, 17].std()
00120     alld[:, 17] = (alld[:, 17] - normsubtract[17]) / normdevide[17] 
00121 
00122     # skip normalized magnitude difference (already small enough)
00123 
00124     # chi2 (devide by 5 * std to avoid overflows due to huge spread
00125     if (newnorm == True):
00126         normsubtract[19] = alld[:, 19].mean()
00127         normdevide[19]   = 5.0 * alld[:, 19].std()
00128     alld[:, 19] = (alld[:, 19] - normsubtract[19]) / normdevide[19] 
00129 
00130     return (alld, allt, allnames, normsubtract, normdevide)
