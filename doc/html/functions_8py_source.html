00040 def deriv2(x, y):
00041     if (len(x) == 0) or (len(x) != len(y)):
00042         return None
00043     sabs = 0.0
00044     for i in xrange(1, len(x) - 1):
00045         f1 = (y[i] - y[i - 1]) / (x[i] - x[i - 1])
00046         f2 = (y[i + 1] - y[i]) / (x[i + 1] - x[i])
00047         d2 = (f2 - f1) / (x[i + 1] - x[i - 1])
00048         sabs += abs(d2)
00049     return sabs / len(x)
